# 问题根因
1. 测试脚本使用alloc_pagecache函数写入文件，但写入操作不会填充page cache ：写入时数据直接进入页缓冲（page buffer），不会被计入page cache统计，只有读取文件时，数据才会被加载到page cache中，因此 memory.stat.cache 始终为0

# 解决方案
修改：memcontrol02.c - alloc_pagecache_50M_check:
1. 确保在cgroup v1环境下使用正确的字段名"cache"
2. 在写入文件后，添加读取操作以填充page cache

```
static void alloc_pagecache_50M_check(void)
{
    const size_t size = MB(50);
    size_t current, file;
    const char *file_key;
    const char *file_key_fmt;

    if (TST_CG_VER_IS_V1(tst_cg, "memory")) {
        file_key = "cache";
        file_key_fmt = "cache %zd";
    } else {
        file_key = "file";
        file_key_fmt = "file %zd";
    }

    fd = SAFE_OPEN(TMPDIR"/tmpfile", O_RDWR | O_CREAT, 0600);

    SAFE_CG_SCANF(cg_child, "memory.current", "%zu", &current);
    tst_res(TINFO, "Created temp file: memory.current=%zu", current);

    alloc_pagecache(fd, size);

    SAFE_CG_SCANF(cg_child, "memory.current", "%zu", &current);
    TST_EXP_EXPR(current >= size,
                 "(memory.current=%zu) >= (size=%zu)", current, size);

    // Read back the file to populate page cache
    SAFE_LSEEK(fd, 0, SEEK_SET);
    char *buf = SAFE_MALLOC(size);
    SAFE_READ(1, fd, buf, size);
    free(buf);

    SAFE_CG_LINES_SCANF(cg_child, "memory.stat", file_key_fmt, &file);
    TST_EXP_EXPR(file > 0, "(memory.stat.%s=%zd) > 0", file_key, file);

    TST_EXP_EXPR(values_close(file, current, file_to_all_error),
                 "(memory.current=%zd) ~= (memory.stat.%s=%zd)",
                 current, file_key, file);

    SAFE_CLOSE(fd);
}
```
